```python
        self.data[today][event_type] += 1
        
        # –°–æ—Ö—Ä–∞–Ω–∏—Ç—å –≤ —Ñ–∞–π–ª
        with open('data/metrics.json', 'w') as f:
            json.dump(self.data, f, indent=2)
```
